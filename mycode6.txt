BFS(Adj, root):
    n = number of nodes in Adj

    color[root] = "gray"
    distance[root] = 0
    parent[root] = null

    create an empty queue Q
    enqueue(Q, root)

    while Q is not empty:
        current = dequeue(Q)

        for j = 0 to n - 1:
            if Adj[current][j] == 1 and color[j] == "white":
                color[j] = "gray"
                distance[j] = distance[current] + 1
                parent[j] = current
                enqueue(Q, j)

        color[current] = "black"
